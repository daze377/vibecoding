\section*{Problem 1}
A circle has radius $r = 3$ cm. Find its area.
\[ A = \pi r^2 = \pi \cdot 3^2 = 9\pi \approx 28.27 \ \mathrm{cm}^2 \]

\section*{Problem 2}
A circle has diameter $d = 10$ cm. Find its area.
\[ r = \frac{d}{2} = 5, \qquad A = \pi r^2 = 25\pi \approx 78.54 \ \mathrm{cm}^2 \]

\section*{Problem 3}
A pizza has area $A = 50.27$ in$^2$. What is its radius?
\[ r = \sqrt{\frac{A}{\pi}} = \sqrt{\frac{50.27}{\pi}} = \sqrt{16} = 4 \ \mathrm{in} \]
